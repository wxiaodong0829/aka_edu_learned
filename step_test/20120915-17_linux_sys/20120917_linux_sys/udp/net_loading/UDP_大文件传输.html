近日小弟做了个linux下用户数据报协议大文件传输程式发上来与大家共勉。
代码在 redhat 9.0 下编译通过。
最大测试无差错传输文件：288M
最大测试传输速度：6.5M/S

可能这里并不需要这种类型的帖子，但希望各位能代小弟转贴，因为网络上很难搜索到这种可以解决丢包问题的代码级实现方法，希望每一个需要的人都能看的到。

  18. void recvUDP(char name[20],int sockfd)
  19. {
  20.         int ret,fd;
  21.         mode_t fdmode = (S_IRUSR|S_IWUSR|S_IRGRP|S_IROTH);
  22.         char mesg[MAX_SIZE];
  23.         fd_set rdset;
  24.         struct timeval tv;
  25.         int rlen,wlen;
  26.         int i;

  28.         fd = open(name,O_RDWR|O_CREAT|O_APPEND,fdmode);
  29.         if(fd == -1)
  30.         {
  31.                 printf("open file %s error:%n",name,strerror(errno));
  32.                 exit(-1);
  33.         }

  35.         for(i=0;;i++)
  36.         {
  37.                 tv.tv_sec = 1;
  38.                 tv.tv_usec = 0;

  40.                 FD_ZERO(&rdset);
  41.                 FD_SET(sockfd,&rdset);
  42.                
  43.                 ret = select(sockfd+1,&rdset,NULL,NULL,&tv);
  44.                 if(ret == -1)
  45.                 {
  46.                         printf("select error %s\n",strerror(errno));
  47.                         exit(-1);
  48.                 }
  49.                 else if(ret==0)
  50.                 {
  51.                         printf("select timeout,continue circle\n");
  52.                         continue;
  53.                 }
  54.                
  55.                 if(FD_ISSET(sockfd,&rdset))
  56.                 {
  57.                         memset(mesg,0,MAX_SIZE);
  58.                         rlen = read(sockfd,mesg,MAX_SIZE);
  59.                         if(rlen <=0 )
  60.                         {
  61.                                 printf("read error %s\n",strerror(errno));
  62.                                 exit(-1);
  63.                         }
  64.                        
  65.                         if(!strcmp(mesg,"end"))
  66.                         {
  67.                                 printf("recv end.\n");
  68.                                 break;
  69.                         }

  71.                         wlen = write(fd,mesg,rlen);
  72.                         if(wlen != rlen )
  73.                         {
  74.                                 printf("write error %s\n",strerror(errno));
  75.                                 exit(-1);
  76.                         }       
  77.                 }
  78.                
  79.                 printf("The %d times write\n",i);
  80.         }

  82.         close(fd);
  83. }

  85. int main(int argc, char *argv[])
  86. {
  87.         int sockfd;
  88.         int r;
  89.         struct sockaddr_in servaddr;

  91.         sockfd = socket(AF_INET,SOCK_DGRAM,0); /*create a socket*/

  93.         /*init servaddr*/
  94.         bzero(&servaddr,sizeof(servaddr));
  95.         servaddr.sin_family = AF_INET;
  96.         servaddr.sin_addr.s_addr = htonl(INADDR_ANY);
  97.         servaddr.sin_port = htons(SERV_PORT);

  99.         /*bind address and port to socket*/
  100.         if(bind(sockfd,(struct sockaddr *)&servaddr,sizeof(servaddr)) == -1)
  101.         {
  102.                 perror("bind error");
  103.                 exit(-1);
  104.         }

  106.         r = fcntl(sockfd, F_GETFL, 0);
  107.         fcntl(sockfd, F_SETFL, r & ~O_NONBLOCK);
  108.        
  109.         recvUDP(argv[1],sockfd);

  111.         return 0;

  113. }

  20. void connectUDP(char name[20],int sockfd,struct sockaddr *pservaddr,socklen_t servlen)
  21. {
  22.                 char buf[MAX_SIZE];
  23.                 fd_set wrset;
  24.                 struct timeval tv;
  25.                 int rlen,wlen;
  26.                 int fd;
  27.                 int ret;
  28.                 int i;

  30.                 if(connect(sockfd,(struct sockaddr *)pservaddr,servlen) == -1)
  31.                 {
  32.                         perror("connet error");
  33.                         exit(1);
  34.                 }
  35.                 else
  36.                         printf("connect server ok!\n");

  38.                 fd = open(name,O_RDONLY);
  39.                 if(fd==-1)
  40.                 {
  41.                         printf("fopen error %s\n",strerror(errno));
  42.                         exit(-1);
  43.                 }
  44.                
  45.                 i=0;
  46.                
  47.                 while(1)
  48.                 {       
  49.                         tv.tv_sec = 1;
  50.                         tv.tv_usec = 0;
  51.                        
  52.                         FD_ZERO(&wrset);
  53.                         FD_SET(sockfd,&wrset);
  54.                        
  55.                         ret = select(sockfd+1,NULL,&wrset,NULL,&tv);
  56.                         if(ret == -1)
  57.                         {
  58.                                 printf("select error %s\n",strerror(errno));
  59.                                 exit(-1);
  60.                         }
  61.                         else if(ret==0)
  62.                         {
  63.                                 printf("select timeout,continue circle\n");
  64.                                 continue;
  65.                         }
  66.                        
  67.                         memset(buf,0,MAX_SIZE);
  68.                         if(FD_ISSET(sockfd,&wrset))
  69.                         {                                       
  70.                                 rlen = read(fd,buf,MAX_SIZE);
  71.                                 if(rlen <0)
  72.                                 {
  73.                                         printf("fread data error %s\n",strerror(errno));
  74.                                         exit(-1);
  75.                                 }
  76.                                 else if(rlen==0)
  77.                                 {
  78.                                         wlen = write(sockfd,"end",3);
  79.                                         if(wlen !=3)
  80.                                         {
  81.                                                 printf("write end error\n",strerror(errno));
  82.                                                 exit(-1);
  83.                                         }

  85.                                         printf("all complete\n");

  87.                                         close(fd);
  88.                                         close(sockfd);
  89.                                         exit(0);
  90.                                 }

  92.                                 wlen = write(sockfd,buf,rlen);
  93.                                 if(wlen != rlen)
  94.                                 {
  95.                                         printf("write data to sockfd error:%s\n",strerror(errno));
  96.                                         exit(-1);
  97.                                 }
  98.                                
  99.                                 i++;
  100.                                
  101.                                 usleep(500);
  102.                                 printf("The %d times read\n",i);
  103.                         }
  104.                 }
  105.         }


  108. int main(int argc ,char *argv[])
